  gl.bindVertexArray(vao);

  const buf = gl.createBuffer();

  gl.bufferData(gl.ARRAY_BUFFER, positions, gl.STATIC_DRAW);
  gl.enableVertexAttribArray(0);
  gl.vertexAttribPointer(0, 2, gl.FLOAT, false, 0, 0);
  gl.bindVertexArray(null);
  
  // uniforms locations
  const u_cameraPos = gl.getUniformLocation(program, 'u_cameraPos');
  const u_cameraForward = ... etc.
  const u_resolution = ...
  const u_fovScale = ...
  const u_time = ...
  
  // initial camera state
  let yaw = 0, pitch = 0, distance = 8.0;
  // compute camera
  const cameraState = { pos: new Float32Array(3), forward: new Float32Array(3), right: new Float32Array(3), up: new Float32Array(3) };
  
  function updateCamera() {
    const horizDist = distance * Math.cos(pitch);
    cameraState.pos[0] = horizDist * Math.cos(yaw);
    cameraState.pos[1] = distance * Math.sin(pitch);
    cameraState.pos[2] = horizDist * Math.sin(yaw);
    // target at origin
    const target = [0,0,0];
    const fwd = new Float32Array([target[0]-cameraState.pos[0], target[1]-cameraState.pos[1], target[2]-cameraState.pos[2]]);
    const len = Math.hypot(fwd[0], fwd[1], fwd[2]);
    fwd[0]/=len; fwd[1]/=len; fwd[2]/=len;
    cameraState.forward = fwd;
    const worldUp = [0,1,0];
    const right = new Float32Array(3);
    // cross(fwd, worldUp)
    right[0] = fwd[1]*worldUp[2] - fwd[2]*worldUp[1];
    right[1] = fwd[2]*worldUp[0] - fwd[0]*worldUp[2];
    right[2] = fwd[0]*worldUp[1] - fwd[1]*worldUp[0];
    // if forward is parallel to worldUp, use different worldUp
    const lenR = Math.hypot(right[0],right[1],right[2]);
    if (lenR < 0.001) {
      const altUp = [0,0,1];
      right[0] = fwd[1]*altUp[2] - fwd[2]*altUp[1];
      right[1] = fwd[2]*altUp[0] - fwd[0]*altUp[2];
      right[2] = fwd[0]*altUp[1] - fwd[1]*altUp[0];
    }
    // normalize right
    right[0]/=Math.hypot(right[0],right[1],right[2]);
    right[1]/=Math.hypot(right[0],right[1],right[2]); // lazy: compute len again
    // simplify: recompute len
    const rlen = Math.hypot(right[0],right[1],right[2]);
    right[0]/=rlen; right[1]/=rlen; right[2]/=rlen;

    const up = [ right[1]*fwd[2] - right[2]*fwd[1], right[2]*fwd[0] - right[0]*fwd[2], right[0]*fwd[1] - right[1]*fwd[0] ];
    // assign
    cameraState.right = right;
    cameraState.up = up;

  const fovScale = Math.tan( (45 * Math.PI/180) / 2 );
  
  function render(time) {

    const h = canvas.height;

    gl.bindVertexArray(vao);
    // set uniforms

    gl.uniform3fv(u_cameraForward, cameraState.forward);
    gl.uniform3fv(u_cameraRight, cameraState.right);
    gl.uniform3fv(u_cameraUp, cameraState.up);
    gl.uniform2f(u_resolution, w, h);
    gl.uniform1f(u_fovScale, fovScale);
    gl.uniform1f(u_time, time);
    gl.drawArrays(gl.TRIANGLE_STRIP, 0, 4);
    gl.bindVertexArray(null);
  }
  
  function setView(yaw_, pitch_, distance_) {
    yaw = yaw_;
    pitch = pitch_;
    distance = distance_;

  }

  }
  
  function dispose() {

    canvas.removeEventListener('pointerup', onPointerUp);
    canvas.removeEventListener('pointercancel', onPointerUp);

    gl.deleteVertexArray(vao);
    gl.deleteBuffer(buf);
