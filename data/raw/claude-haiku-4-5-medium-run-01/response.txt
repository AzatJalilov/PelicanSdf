```javascript
export function createPelicanSdf(canvas) {
  canvas.tabIndex = 1;
  const gl = canvas.getContext('webgl2');
  if (!gl) throw new Error('WebGL2 required');

  const vertSrc = `#version 300 es
    in vec2 pos;
    void main() { gl_Position = vec4(pos, 0.0, 1.0); }
  `;

  const fragSrc = `#version 300 es
    precision highp float;
    uniform vec3 camPos, camRight, camUp, camFwd;
    uniform vec2 resolution;
    uniform float time;
    out vec4 outColor;

    float sdf(vec3 p);
    
    void main() {
      vec2 uv = gl_FragCoord.xy / resolution;
      uv = uv * 2.0 - 1.0;
      uv.x *= resolution.x / resolution.y;
      
      vec3 ro = camPos;
      vec3 rd = normalize(uv.x * camRight + uv.y * camUp + 2.0 * camFwd);
      
      vec3 p = ro;
      float traveled = 0.0;
      for (int i = 0; i < 220; i++) {
        float dist = sdf(p);
        if (dist < 0.0008 || traveled > 120.0) break;
        p += rd * dist;
        traveled += dist;
      }
      
      if (sdf(p) < 0.01) {
        float h = 0.0008;
        vec3 n = normalize(vec3(
          sdf(p+vec3(h,0,0))-sdf(p-vec3(h,0,0)),
          sdf(p+vec3(0,h,0))-sdf(p-vec3(0,h,0)),
          sdf(p+vec3(0,0,h))-sdf(p-vec3(0,0,h))
        ));
        vec3 l = normalize(vec3(1.2, 1.5, 0.8));
        float diff = max(0.0, dot(n, l));
        vec3 refl = reflect(l, n);
        float spec = pow(max(0.0, dot(-rd, refl)), 16.0);
        
        vec3 col = vec3(0.85, 0.72, 0.55);
        if (sdf(p + n*0.05) > 0.1) col = vec3(0.3, 0.4, 0.2);
        col = col * (0.25 + 0.75 * diff) + 0.3 * spec;
        outColor = vec4(col, 1.0);
      } else {
        vec3 sky = mix(vec3(0.7, 0.85, 1.0), vec3(0.9, 0.95, 1.0), pow(max(0.0, rd.y + 0.3), 2.0));
        outColor = vec4(sky, 1.0);
      }
    }
    
    mat3 rx(float a) { float c=cos(a), s=sin(a); return mat3(1,0,0, 0,c,-s, 0,s,c); }
    mat3 ry(float a) { float c=cos(a), s=sin(a); return mat3(c,0,s, 0,1,0, -s,0,c); }
    mat3 rz(float a) { float c=cos(a), s=sin(a); return mat3(c,-s,0, s,c,0, 0,0,1); }
    
    float sdSphere(vec3 p, float r) { return length(p) - r; }
    float sdBox(vec3 p, vec3 b) {
      vec3 q = abs(p) - b;
      return length(max(q,0.0)) + min(max(q.x,max(q.y,q.z)),0.0);
    }
    float sdCylinder(vec3 p, float r, float h) {
      vec2 d = abs(vec2(length(p.xz), p.y)) - vec2(r, h);
      return min(max(d.x,d.y),0.0) + length(max(d,0.0));
    }
    float sdCapsule(vec3 p, vec3 a, vec3 b, float r) {
      vec3 pa = p - a, ba = b - a;
      float h = clamp(dot(pa, ba) / dot(ba, ba), 0.0, 1.0);
      return length(pa - ba * h) - r;
    }
    float sdTorus(vec3 p, vec2 t) {
      vec2 q = vec2(length(p.xz) - t.x, p.y);
      return length(q) - t.y;
    }
    
    float sdf(vec3 p) {
      float d = 1e6;
      
      // Bicycle frame - main triangle
      d = min(d, sdCapsule(p, vec3(-2.0, 0.3, 0.0), vec3(0.4, -0.9, 0.0), 0.085));
      d = min(d, sdCapsule(p, vec3(0.4, -0.9, 0.0), vec3(-2.0, -0.9, 0.0), 0.085));
      d = min(d, sdCapsule(p, vec3(-2.0, 0.3, 0.0), vec3(-2.0, -0.9, 0.0), 0.085));
      
      // Seat tube
      d = min(d, sdCapsule(p, vec3(0.4, -0.9, 0.0), vec3(-0.1, 0.25, 0.0), 0.07));
      
      // Handlebar stem
      d = min(d, sdCapsule(p, vec3(0.4, -0.35, 0.0), vec3(0.6, -0.5, 0.0), 0.055));
      
      // Handlebars
      d = min(d, sdCapsule(p, vec3(0.6, -0.5, -0.42), vec3(0.6, -0.5, 0.42), 0.065));
      d = min(d, sdCapsule(p, vec3(0.55, -0.5, -0.4), vec3(0.65, -0.35, -0.4), 0.04));
      d = min(d, sdCapsule(p, vec3(0.55, -0.5, 0.4), vec3(0.65, -0.35, 0.4), 0.04));
      
      // Seat
      d = min(d, sdBox(p - vec3(-0.1, 0.35, 0.0), vec3(0.28, 0.085, 0.14)));
      
      // Wheels
      float wheelL = sdTorus(p - vec3(-2.0, -0.9, -0.85), vec2(0.65, 0.13));
      float wheelR = sdTorus(p - vec3(-2.0, -0.9, 0.85), vec2(0.65, 0.13));
      d = min(d, min(wheelL, wheelR));
      
      // Wheel spokes
      vec3 ws = (p - vec3(-2.0, -0.9, 0.0)) * ry(time * 2.8);
      d = min(d, sdCylinder(ws, 0.035, 0.12));
      d = min(d, sdCylinder(ws * rz(1.5708), 0.035, 0.12));
      d = min(d, sdCylinder(ws * rz(3.14159), 0.035, 0.12));
      d = min(d, sdCylinder(ws * rz(4.71239), 0.035, 0.12));
      
      // Pedal crank
      vec3 cp = (p - vec3(-0.3, -0.9, 0.0)) * rz(time * 2.2);
      d = min(d, sdCapsule(cp, vec3(0,0,0), vec3(0.38, 0.0, 0.0), 0.06));
      d = min(d, sdSphere(cp - vec3(0.38, 0.0, 0.0), 0.076));
      
      // Chain guard
      d = min(d, sdCapsule(p, vec3(-0.3, -0.5, -0.2), vec3(-0.3, -0.5, 0.2), 0.09));
      
      // Pelican body
      d = min(d, sdBox(p - vec3(-0.15, 0.35, 0.0), vec3(0.42, 0.30, 0.24)));
      
      // Pelican head
      d = min(d, sdSphere(p - vec3(0.35, 0.80, 0.0), 0.24));
      
      // Pelican beak
      vec3 bk = p - vec3(0.58, 0.82, 0.0);
      d = min(d, sdCapsule(bk, vec3(0,0,0), vec3(0.88, -0.12, 0.0), 0.095));
      d = min(d, sdCapsule(bk, vec3(0.3, -0.06, -0.08), vec3(0.3, -0.06, 0.08), 0.055));
      
      // Throat pouch
      d = min(d, sdSphere(p - vec3(0.65, 0.48, 0.0), 0.16));
      
      // Eye
      d = min(d, sdSphere(p - vec3(0.45, 0.88, 0.18), 0.075));
      
      // Pelican wings (folded along sides)
      d = min(d, sdBox(p - vec3(-0.35, 0.35, -0.4), vec3(0.22, 0.38, 0.14)));
      d = min(d, sdBox(p - vec3(-0.35, 0.35, 0.4), vec3(0.22, 0.38, 0.14)));
      
      // Wing tips
      d = min(d, sdCapsule(p, vec3(-0.6, 0.2, -0.5), vec3(-0.8, 0.0, -0.65), 0.08));
      d = min(d, sdCapsule(p, vec3(-0.6, 0.2, 0.5), vec3(-0.8, 0.0, 0.65), 0.08));
      
      // Pelican legs (reaching down to pedals)
      d = min(d, sdCapsule(p, vec3(-0.25, 0.1, -0.18), vec3(-0.25, -0.55, -0.18), 0.048));
      d = min(d, sdCapsule(p, vec3(0.1, 0.1, 0.22), vec3(0.1, -0.55, 0.22), 0.048));
      
      // Feet
      d = min(d, sdSphere(p - vec3(-0.25, -0.58, -0.18), 0.063));
      d = min(d, sdSphere(p - vec3(0.1, -0.58, 0.22), 0.063));
      
      // Tail feathers
      d = min(d, sdBox(p - vec3(-0.65, 0.25, 0.0), vec3(0.15, 0.25, 0.18)));
      
      return d;
    }
  `;

  const compileShader = (src, type) => {
    const sh = gl.createShader(type);
    gl.shaderSource(sh, src);
    gl.compileShader(sh);
    if (!gl.getShaderParameter(sh, gl.COMPILE_STATUS)) {
      throw new Error('Shader compile: ' + gl.getShaderInfoLog(sh));
    }
    return sh;
  };

  const prog = gl.createProgram();
  gl.attachShader(prog, compileShader(vertSrc, gl.VERTEX_SHADER));
  gl.attachShader(prog, compileShader(fragSrc, gl.FRAGMENT_SHADER));
  gl.linkProgram(prog);
  if (!gl.getProgramParameter(prog, gl.LINK_STATUS)) {
    throw new Error('Program link: ' + gl.getProgramInfoLog(prog));
  }
  gl.useProgram(prog);

  const vao = gl.createVertexArray();
  gl.bindVertexArray(vao);
  const buf = gl.createBuffer();
  gl.bindBuffer(gl.ARRAY_BUFFER, buf);
  gl.bufferData(gl.ARRAY_BUFFER, new Float32Array([-1,-1,1,-1,-1,1,1,1]), gl.STATIC_DRAW);
  const loc = gl.getAttribLocation(prog, 'pos');
  gl.enableVertexAttribArray(loc);
  gl.vertexAttribPointer(loc, 2, gl.FLOAT, false, 0, 0);

  let yaw = -0.8, pitch = 0.35, dist = 4.2;

  const updateCam = () => {
    const cy = Math.cos(yaw), sy = Math.sin(yaw);
    const cp = Math.cos(pitch), sp = Math.sin(pitch);
    
    const pos = [cy * cp * dist, sp * dist, sy * cp * dist];
    const fwd = [-cy * cp, -sp, -sy * cp];
    const right = [sy, 0, -cy];
    const up = [-sp * sy, cp, sp * cy];
    
    gl.uniform3f(gl.getUniformLocation(prog, 'camPos'), pos[0], pos[1], pos[2]);
    gl.uniform3f(gl.getUniformLocation(prog, 'camRight'), right[0], right[1], right[2]);
    gl.uniform3f(gl.getUniformLocation(prog, 'camUp'), up[0], up[1], up[2]);
    gl.uniform3f(gl.getUniformLocation(prog, 'camFwd'), fwd[0], fwd[1], fwd[2]);
  };

  let dragging = false, lastX = 0, lastY = 0, lastTouchDist = 0;
  
  const onMouseDown = (e) => {
    dragging = true;
    lastX = e.clientX;
    lastY = e.clientY;
  };

  const onMouseMove = (e) => {
    if (dragging) {
      yaw += (e.clientX - lastX) * 0.008;
      pitch += (e.clientY - lastY) * 0.008;
      pitch = Math.max(-1.48, Math.min(1.48, pitch));
      lastX = e.clientX;
      lastY = e.clientY;
    }
  };

  const onMouseUp = () => { dragging = false; };

  const onWheel = (e) => {
    e.preventDefault();
    dist *= Math.exp(e.deltaY * 0.0008);
    dist = Math.max(1.4, Math.min(16, dist));
  };

  const onKeyDown = (e) => {
    if (e.key === 'ArrowLeft') yaw -= 0.12;
    if (e.key === 'ArrowRight') yaw += 0.12;
    if (e.key === 'ArrowUp') pitch = Math.min(1.48, pitch + 0.12);
    if (e.key === 'ArrowDown') pitch = Math.max(-1.48, pitch - 0.12);
    if (e.key === '+' || e.key === '=') dist *= 0.93;
    if (e.key === '-') dist *= 1.075;
    dist = Math.max(1.4, Math.min(16, dist));
  };

  const onTouchStart = (e) => {
    if (e.touches.length === 1) {
      dragging = true;
      lastX = e.touches[0].clientX;
      lastY = e.touches[0].clientY;
    } else if (e.touches.length === 2) {
      const dx = e.touches[0].clientX - e.touches[1].clientX;
      const dy = e.touches[0].clientY - e.touches[1].clientY;
      lastTouchDist = Math.sqrt(dx*dx + dy*dy);
    }
  };

  const onTouchMove = (e) => {
    e.preventDefault();
    if (e.touches.length === 1 && dragging) {
      yaw += (e.touches[0].clientX - lastX) * 0.008;
      pitch += (e.touches[0].clientY - lastY) * 0.008;
      pitch = Math.max(-1.48, Math.min(1.48, pitch));
      lastX = e.touches[0].clientX;
      lastY = e.touches[0].clientY;
    } else if (e.touches.length === 2) {
      const dx = e.touches[0].clientX - e.touches[1].clientX;
      const dy = e.touches[0].clientY - e.touches[1].clientY;
      const d = Math.sqrt(dx*dx + dy*dy);
      if (lastTouchDist > 0) {
        dist *= Math.max(0.85, Math.min(1.15, lastTouchDist / d));
        dist = Math.max(1.4, Math.min(16, dist));
      }
      lastTouchDist = d;
    }
  };

  const onTouchEnd = () => { dragging = false; lastTouchDist = 0; };

  canvas.addEventListener('mousedown', onMouseDown);
  canvas.addEventListener('mousemove', onMouseMove);
  canvas.addEventListener('mouseup', onMouseUp);
  canvas.addEventListener('wheel', onWheel, { passive: false });
  canvas.addEventListener('keydown', onKeyDown);
  canvas.addEventListener('touchstart', onTouchStart, { passive: true });
  canvas.addEventListener('touchmove', onTouchMove, { passive: false });
  canvas.addEventListener('touchend', onTouchEnd, { passive: true });

  const render = (timeSeconds) => {
    updateCam();
    gl.viewport(0, 0, canvas.width, canvas.height);
    gl.clearColor(0.8, 0.9, 1.0, 1.0);
    gl.clear(gl.COLOR_BUFFER_BIT);
    gl.uniform2f(gl.getUniformLocation(prog, 'resolution'), canvas.width, canvas.height);
    gl.uniform1f(gl.getUniformLocation(prog, 'time'), timeSeconds);
    gl.drawArrays(gl.TRIANGLE_STRIP, 0, 4);
  };

  const setView = (newYaw, newPitch, newDist) => {
    yaw = newYaw;
    pitch = newPitch;
    dist = newDist;
  };

  const getView = () => ({ yaw, pitch, distance: dist });

  const dispose = () => {
    canvas.removeEventListener('mousedown', onMouseDown);
    canvas.removeEventListener('mousemove', onMouseMove);
    canvas.removeEventListener('mouseup', onMouseUp);
    canvas.removeEventListener('wheel', onWheel);
    canvas.removeEventListener('keydown', onKeyDown);
    canvas.removeEventListener('touchstart', onTouchStart);
    canvas.removeEventListener('touchmove', onTouchMove);
    canvas.removeEventListener('touchend', onTouchEnd);
    gl.deleteProgram(prog);
    gl.deleteBuffer(buf);
    gl.deleteVertexArray(vao);
  };

  return { render, setView, getView, dispose };
}
```
