```glsl
float smin(float a,float b,float k){
  float h = clamp(0.5+0.5*(b-a)/k, 0.0, 1.0);
  return mix(b,a,h) - k*h*(1.0-h);
}
float sph(vec3 p,float r){ return length(p)-r; }
float ell(vec3 p, vec3 r){
  float k0 = length(p/r);
  float k1 = length(p/(r*r));
  return k0*(k0-1.0)/max(k1,1e-6);
}
float cap(vec3 p, vec3 a, vec3 b, float r){
  vec3 pa=p-a, ba=b-a;
  float h = clamp(dot(pa,ba)/dot(ba,ba),0.0,1.0);
  return length(pa-ba*h)-r;
}
float rcone(vec3 p, vec3 a, vec3 b, float r1, float r2){
  vec3 ba=b-a;
  float l2=dot(ba,ba), rr=r1-r2, a2=l2-rr*rr, il2=1.0/l2;
  vec3 pa=p-a;
  float y=dot(pa,ba), z=y-l2;
  vec3 xp = pa*l2 - ba*y;
  float x2=dot(xp,xp), y2=y*y*l2, z2=z*z*l2;
  float k = sign(rr)*rr*rr*x2;
  if(sign(z)*a2*z2 > k) return sqrt(x2+z2)*il2 - r2;
  if(sign(y)*a2*y2 < k) return sqrt(x2+y2)*il2 - r1;
  return (sqrt(x2*a2*il2)+y*rr)*il2 - r1;
}
float torZ(vec3 p, vec2 t){
  vec2 q = vec2(length(p.xy)-t.x, p.z);
  return length(q)-t.y;
}
float cylZ(vec3 p, float r, float h){
  vec2 d = abs(vec2(length(p.xy), p.z)) - vec2(r,h);
  return min(max(d.x,d.y),0.0)+length(max(d,0.0));
}
float boxf(vec3 p, vec3 b){
  vec3 q = abs(p)-b;
  return length(max(q,0.0))+min(max(q.x,max(q.y,q.z)),0.0);
}
vec3 rotZ(vec3 p,float a){ float c=cos(a),s=sin(a); return vec3(c*p.x+s*p.y, -s*p.x+c*p.y, p.z); }
vec3 rotY(vec3 p,float a){ float c=cos(a),s=sin(a); return vec3(c*p.x+s*p.z, p.y, -s*p.x+c*p.z); }
```
